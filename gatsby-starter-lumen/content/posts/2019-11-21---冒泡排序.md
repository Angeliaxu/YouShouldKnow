function Bubble(arr) {
    var temp;
    for(let i = 0; i < arr.length; i++) {
        for(let j = 0; j < arr.length-i-1; j++) {
            if(arr[j] > arr[j+1]) {
                temp = arr[j+1];
                arr[j+1] = arr[j];
                arr[j] =temp
            }
        }
    }
    return arr;
}
const arr = [2, 5, 7, 9, 3, 4, 1];
Bubble(arr);
console.log(arr)
